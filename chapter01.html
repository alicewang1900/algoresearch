Sample Problem 1: Count Vowels

Given a string, count how many vowels it contains.

Count Vowels (Python)

def count_vowels(s):
    return sum(1 for c in s.lower() if c in "aeiou")

print(count_vowels(input()))

Count Vowels (C++)

#include 
#include 
using namespace std;

int main() {
    string s;
    getline(cin, s);
    int count = 0;
    for (char c : s) {
        c = tolower(c);
        if (c == 'a' || c == 'e' || c == 'i' || c == 'o' || c == 'u') count++;
    }
    cout << count << endl;
    return 0;
}

Sample Problem 2: Simulate Robot

Move a robot on a grid based on L/R/U/D commands.

Robot Movement (Python)

x = y = 0
for c in input():
    if c == 'L': x -= 1
    elif c == 'R': x += 1
    elif c == 'U': y += 1
    elif c == 'D': y -= 1
print(x, y)

Robot Movement (C++)

#include 
using namespace std;

int main() {
    string s;
    cin >> s;
    int x = 0, y = 0;
    for (char c : s) {
        if (c == 'L') x--;
        else if (c == 'R') x++;
        else if (c == 'U') y++;
        else if (c == 'D') y--;
    }
    cout << x << " " << y << endl;
    return 0;
}
